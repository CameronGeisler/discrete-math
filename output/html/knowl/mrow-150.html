\begin{align*} 1^2 + 2^2 + \cdots + k^2 + (k + 1)^2 \amp = \sum_{i=1}^{k+1} i^2\\ \amp = \sum_{i=1}^k i^2 + (k + 1)^2 \amp\amp \text{separating off the final term}\\ \amp = \frac{k(k + 1)(2k + 1)}{6} + (k + 1)^2 \amp\amp \text{by the induction hypothesis}\\ \amp = \frac{k(k + 1)(2k + 1)}{6} + \frac{6(k + 1)^2}{6}\\ \amp = \frac{(k + 1) \brac{k(2k + 1) + 6(k + 1)}}{6} \amp\amp \text{combining fractions, factoring out $(k + 1)$}\\ \amp = \frac{(k + 1)(2k^2 + 7k + 6)}{6} \amp\amp \text{simplifying}\\ \amp = \frac{(k + 1)(k + 2)(2k + 3)}{6} \amp\amp \text{factoring the quadratic expression} \end{align*}
