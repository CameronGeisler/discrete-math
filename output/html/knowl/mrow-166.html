\begin{align*} 1^3 + 2^3 + 3^3 + \dots + (n - 1)^3 + n^3 + (n + 1)^3 \amp = \sum_{i=1}^{n+1} i^3\\ \amp = \sum_{i=1}^n i^3 + (n + 1)^3 \amp\amp \text{separating off the final term}\\ \amp = \brac{\frac{n(n + 1)}{2}}^2 + (n + 1)^3 \amp\amp \text{by the induction hypothesis}\\ \amp = \frac{n^2(n + 1)^2}{4} + (n + 1)^3\\ \amp = \frac{n^2(n + 1)^2 + 4(n + 1)^3}{4}\\ \amp = \frac{(n + 1)^2 (n^2 + 4n + 4)}{4} \amp\amp \text{factoring $(n + 1)^2$ from both terms}\\ \amp = \frac{(n + 1)^2(n + 2)^2}{4} \end{align*}
